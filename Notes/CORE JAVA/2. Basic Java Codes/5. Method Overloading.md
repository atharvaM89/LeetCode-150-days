## 4. Greatest Common Divisor of Two Numbers

```java
package pkg4.greatest.common.divisor.of.pkg2.number;
import java.util.*;
public class GreatestCommonDivisorOf2Number 
{
    static int Gcd(int m, int n)
    {
        while(m!=n)
        {
            if(m>n)
                m=m-n;
            else
                n=n-m;
        }
        return m;
        
    }
    public static void main(String[] args) 
    {
        System.out.println("Enter two Number =");
        Scanner sc=new Scanner(System.in);
        int m=sc.nextInt();
        int n=sc.nextInt();
        
        System.out.println("The GCD of Entered Number is-");
        System.out.println(Gcd(m,n));
        
        
    }
    
}

```

## 5. Highest Element in Array

```java
package pkg5.highest.element.in.array;
public class HighestElementInArray 
{
    static int max(int A[])
    {
        int max=A[0];
        int i;
        for(i=0;i<A.length-1;i++)
        {
            if(A[i]>max)
                max=A[i];
        }
        return A[i];
        
    }
    public static void main(String[] args) 
    {
        int A[]={1,2,3,4,9};
```

## 7. Reverse Number OverLoading

```java
package pkg7.reverse.number.overloading;
import java.util.*;
public class ReverseNumberOverloading 
{
    static int reverse(int n)
    {
        
        int rev=0;
        while(n!=0)
        {
            rev=(rev*10)+n;
            n=n/10;
            
        }
        return rev;
            
        
    }
        
    public static void main(String[] args) 
    {
        System.out.println("Enter the number which you want to reverse");
        Scanner sc=new Scanner(System.in);
        int j=sc.nextInt();
        System.out.println("The Reverse of the Number is ="+reverse(j));
    }
    
}
```

## 11. Maximum of Number Using Variable Argument

```java
package max.of.number.using.var.argument;
public class MaxOfNumberUsingVarArgument 
{
    static int max(int ...x)
    {
        if(x.length==0)
            return Integer.MIN_VALUE;
        int max=x[0];
        for(int i=1;i<x.length; i++)
        
            if(x[i]>max)
                max=x[i];
            return max;
            
        
    }
    public static void main(String[] args) 
    {
        System.out.println(max());
        System.out.println(max(10,20));
        System.out.println(max(10,20,30));
        System.out.println(max(10,20,30,40));
        System.out.println(max(10,20,30,40,50));
        
    }
    
}

```
